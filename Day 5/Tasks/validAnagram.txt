class Solution {
    public boolean isAnagram(String s, String t) {
        int l=s.length();
        int tl=t.length();
        if(l!=tl){
            return false;
        }
        char [] s1=s.toCharArray();
        char [] t1=t.toCharArray();
        Arrays.sort(s1);
        Arrays.sort(t1);
        return Arrays.equals(s1,t1);
        
    }
}
